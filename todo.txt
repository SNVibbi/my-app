4. React props

/* Passing a react prp as an attribute */
<Car brand="Ford" />

/* Component receiving the prop */
function Car(props) {
    return <h2>I am a { props.brand }!</h2>;
}

/* You can destructure props */
function Car({ brand }) {
    return <h2>I am a { brand }!</h2>;
}
